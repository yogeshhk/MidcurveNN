\begin{frame}[fragile]\frametitle{Dense Network Architecture}

	\begin{itemize}
	\item Gradual dimension reduction
	\item Multiple dense layers with ReLU activation
	\item Symmetric encoder-decoder
	\end{itemize}

\begin{lstlisting}
# Dense Encoder-Decoder (pseudocode)
x = Input(shape=(10000,))          # 100x100 flattened
x = Dense(4096, activation='relu')(x)
x = Dense(1024, activation='relu')(x)
x = Dense(256,  activation='relu')(x)   # bottleneck
x = Dense(1024, activation='relu')(x)
x = Dense(4096, activation='relu')(x)
out = Dense(10000, activation='sigmoid')(x)
model.compile(optimizer='adam', loss='binary_crossentropy')
\end{lstlisting}

\begin{frame}[fragile]\frametitle{Convolutional Network Architecture}

	\begin{itemize}
	\item 4 convolutional blocks with skip connections
	\item Batch normalization + dynamic learning rate
	\end{itemize}

\begin{lstlisting}
# CNN Encoder-Decoder (pseudocode)
x  = Input(shape=(128, 128, 1))
# Encoder
e1 = Conv2D(32, (3,3), activation='relu', padding='same')(x)
e2 = Conv2D(64, (3,3), activation='relu')(MaxPool2D()(e1))
e3 = Conv2D(128,(3,3), activation='relu')(MaxPool2D()(e2))
e4 = Conv2D(256,(3,3), activation='relu')(MaxPool2D()(e3))
# Decoder (with skip connections)
d3 = Conv2DTranspose(128,(2,2),strides=2)(e4)
d3 = BatchNormalization()(concatenate([d3, e3]))
d2 = Conv2DTranspose(64, (2,2),strides=2)(d3)
d1 = Conv2DTranspose(32, (2,2),strides=2)(d2)
out = Conv2D(1, (1,1), activation='sigmoid')(d1)
\end{lstlisting}
\end{frame}
